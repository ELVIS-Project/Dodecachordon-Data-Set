# Example Metadata

The following example shows what should be included in the meta data of the music xml file:
```
<?xml version="1.0" encoding="UTF-8"?>
<!DOCTYPE score-partwise PUBLIC "-//Recordare//DTD MusicXML 3.0 Partwise//EN" "http://www.musicxml.org/dtds/partwise.dtd">
<score-partwise>
  <work>
    <work-number>JPV 157</work-number>
    <work-title>Mangofish go Mountain Climbing</work-title>
    </work>
  <movement-number>1</movement-number>
  <movement-title>Prelude</movement-title>
  <identification>
    <creator type="arranger">Henricus Montibus</creator>
    <creator type="composer">Jakobus Piscatoris</creator>
    <creator type="lyricist">Willibald von der Wolfensteingrube</creator>
    <creator type="translator">Maximilian Ehrhardt</creator>
    <rights>1534</rights>
    <encoding>
      <software>MuseScore 2.0.2</software>
      <encoding-date>2016-03-21</encoding-date>
      <supports element="accidental" type="yes"/>
      <supports element="beam" type="yes"/>
      <supports element="print" attribute="new-page" type="no"/>
      <supports element="print" attribute="new-system" type="no"/>
      <supports element="stem" type="yes"/>
    </encoding>
    <source>De Musica, pisces, et ascensus montium</source>
    <miscellaneous>
      <miscellaneous-field name="description">
        De Musica is an obscure treatise written by an anonymous author from Alto Adige.
      </miscellaneous-field>
    </miscellaneous>
  </identification>
```
