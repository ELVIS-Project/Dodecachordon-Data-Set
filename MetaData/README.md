# Example Metadata

The following example shows what should be included in the meta data of the music xml file:
```
<?xml version="1.0" encoding="UTF-8"?>
<!DOCTYPE score-partwise PUBLIC "-//Recordare//DTD MusicXML 3.0 Partwise//EN" "http://www.musicxml.org/dtds/partwise.dtd">
<score-partwise>
    <work>
        <work-number>Dodecachordon, Book 2</work-number>
        <work-title>Ave praeclara</work-title>
    </work>
    <movement-number>p. 159-161 : p. 192-194</movement-number>
    <movement-title>Ave praeclara</movement-title>
    <identification>
        <creator type="arranger">Glarean</creator>
        <creator type="composer">Anonymous</creator>
        <creator type="lyricist">Prosade B. Virgine</creator>
        <creator type="poet">Ionian/Hypoionian</creator>
        <encoding>
            <software>MuseScore 2.0.3.1</software>
            <encoding-date>2017-01-23</encoding-date>
            <supports element="accidental" type="yes"/>
            <supports element="beam" type="yes"/>
            <supports element="print" attribute="new-page" type="no"/>
            <supports element="print" attribute="new-system" type="no"/>
            <supports element="stem" type="yes"/>
        </encoding>
        ...
    </identification>
```
